/*
Definition for a binary tree node.
struct TreeNode 
//定义二叉树的节点
{
	int val;//结点存储的整数值
    TreeNode *left;//指向左儿子的指针
    TreeNode *right;//指向右儿子的指针
    //三个构造函数
    TreeNode() : val(0), left(nullptr), right(nullptr) {}
    //无参数构造函数，将val初始化为0，左右指针均为nullptr
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
    //只传入一个值
    TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
    //可直接指定节点的值和左右子节点
};
*/
void print(TreeNode *node,vector<int> &res)
{
	if(!node) return;
	else
	{
		print(node->left,res);
		res.push_back(node->val);
		print(node->right,res);
	}
}

        	vector<int> res;

//判断一个二叉树是否对称
//类似的题：Leetcode 相同的树，对称二叉树
bool check(TreeNode *p,TreeNode *q)
{
	if(!p&&!q) return true;
	if(!p||!q) return false;
	return (p->val==q->val)&&check(p->left,q->right)&&check(p->right,q->left);
}
class Solution {
public:
    bool isSymmetric(TreeNode* root) 
    {
        return check(root->left,root->right);
    }
};

**三种遍历方式**

```c++
#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
int lchild[N],rchild[N],n;
void printfront(int x)
{
	if(!lchild[x]&&!rchild[x]) cout<<x<<" ";
	else
	{
		cout<<x<<" ";
		if(lchild[x]) printfront(lchild[x]);
		if(rchild[x]) printfront(rchild[x]);
	}
}
void printmiddle(int x)
{
	if(!lchild[x]&&!rchild[x]) cout<<x<<" ";
	else
	{
		if(lchild[x]) printmiddle(lchild[x]);
		cout<<x<<" ";
		if(rchild[x]) printmiddle(rchild[x]);
	}
}
void printback(int x)
{
	if(!lchild[x]&&!rchild[x]) cout<<x<<" ";
	else
	{
		if(lchild[x]) printback(lchild[x]);
		if(rchild[x]) printback(rchild[x]);
		cout<<x<<" ";
	}
}
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++) cin>>lchild[i]>>rchild[i];
	printfront(1);
	cout<<endl;
	printmiddle(1);
	cout<<endl;
	printback(1);
	return 0;
}
```
